73 QJsonValue QJsonSerializer::serializeVariant(int propertyType, const QVariant &value) const
74 {
75  auto convertValue = value;
76  if((propertyType == QVariant::List) ||
77  (convertValue.canConvert(QVariant::List) && convertValue.convert(QVariant::List))) {
78  return serializeList(propertyType, value.toList());
79  } else {
80  convertValue = value;
81  auto flags = QMetaType::typeFlags(propertyType);
82 
83  if(flags.testFlag(QMetaType::IsGadget)) {
84  auto metaObject = QMetaType::metaObjectForType(propertyType);
85  return serializeGadget(value.constData(), metaObject);
86  } else if(flags.testFlag(QMetaType::PointerToQObject) ||
87  (convertValue.canConvert(QMetaType::QObjectStar) && convertValue.convert(QMetaType::QObjectStar))) {
88  auto object = convertValue.value<QObject*>();
89  if(object)
90  return serializeObject(object);
91  else
92  return QJsonValue::Null;
93  } else
94  return serializeValue(propertyType, value);
95  }
96 }

98 QJsonObject QJsonSerializer::serializeObject(const QObject *object) const
99 {
100  auto meta = object->metaObject();
101 
102  QJsonObject jsonObject;
103  //go through all properties and try to serialize them
104  auto i = QObject::staticMetaObject.indexOfProperty("objectName");
105  if(!d->keepObjectName)
106  i++;
107  for(; i < meta->propertyCount(); i++) {
108  auto property = meta->property(i);
109  if(property.isStored()) {
110  QJsonValue value;
111  if(property.isEnumType())
112  value = serializeEnum(property.enumerator(), property.read(object));
113  else
114  value = serializeVariant(property.userType(), property.read(object));
115  jsonObject[QString::fromUtf8(property.name())] = value;
116  }
117  }
118 
119  return jsonObject;
120 }

166 {
167  if(!value.isValid())
168  return QJsonValue::Null;
169  else {
170  if(value.userType() == QMetaType::QJsonValue)//value needs special treatment
171  return value.value<QJsonValue>();
172 
173  auto json = QJsonValue::fromVariant(value);
174  if(json.isNull()) {
175  if(propertyType == QMetaType::QDate ||
176  propertyType == QMetaType::QTime ||
177  propertyType == QMetaType::QDateTime ||
178  value.userType() == QMetaType::QDate ||
179  value.userType() == QMetaType::QTime ||

182  else

192 QVariant QJsonSerializer::deserializeVariant(int propertyType, const QJsonValue &value, QObject *parent) const
193 {
194  QVariant variant;
195  if(propertyType == QMetaType::QJsonValue)//special case: target type is a json value!
196  variant = QVariant::fromValue(value);
197  else if(value.isArray()) {
198  if(propertyType == QMetaType::QJsonArray)//special case: target type is a json array!
199  variant = QVariant::fromValue(value.toArray());
200  else
201  variant = deserializeList(propertyType, value.toArray(), parent);
202  } else if(value.isObject() || value.isNull()) {
203  auto flags = QMetaType::typeFlags(propertyType);
204 
205  if(propertyType == QMetaType::QJsonObject)//special case: target type is a json object!
206  variant = QVariant::fromValue(value.toObject());
207  else if(flags.testFlag(QMetaType::IsGadget)) {
208  if(!value.isNull()) {
209  QVariant gadget(propertyType, nullptr);
210  deserializeGadget(value.toObject(), propertyType, gadget.data());
211  variant = gadget;
212  }
213  } else if(flags.testFlag(QMetaType::PointerToQObject)) {
214  if(value.isNull())
215  variant = QVariant::fromValue<QObject*>(nullptr);
216  else {
217  auto object = deserializeObject(value.toObject(), QMetaType::metaObjectForType(propertyType), parent);
218  variant = QVariant::fromValue(object);
219  }
220  } else
221  variant = deserializeValue(propertyType, value);
222  } else
223  variant = deserializeValue(propertyType, value);
224 
225  if(propertyType != QMetaType::UnknownType) {
226  auto vType = variant.typeName();
227  if(variant.canConvert(propertyType) && variant.convert(propertyType))
228  return variant;
229  else if(d->allowNull && value.isNull())
230  return QVariant();
231  else {
232  throw QJsonDeserializationException(QByteArray("Failed to convert deserialized variant of type ") +
233  (vType ? vType : "<unknown>") +
234  QByteArray(" to property type ") +
235  QMetaType::typeName(propertyType));
236  }
237  } else
238  return variant;
239 }

241 QObject *QJsonSerializer::deserializeObject(const QJsonObject &jsonObject, const QMetaObject *metaObject, QObject *parent) const
242 {
243  //try to construct the object
244  auto object = metaObject->newInstance(Q_ARG(QObject*, parent));
245  if(!object)
246  throw QJsonDeserializationException(QByteArray("Failed to construct object of type ") +
247  metaObject->className() +
248  QByteArray(" (Does the constructor \"Q_INVOKABLE class(QObject*);\" exist?)"));
249 
250  //now deserialize all json properties
251  for(auto it = jsonObject.constBegin(); it != jsonObject.constEnd(); it++) {
252  auto propIndex = metaObject->indexOfProperty(qUtf8Printable(it.key()));
253  QVariant value;
254  if(propIndex != -1) {
255  auto property = metaObject->property(propIndex);
256  if(property.isEnumType())
257  value = deserializeEnum(property.enumerator(), it.value());
258  else
259  value = deserializeVariant(property.userType(), it.value(), object);
260  } else
261  deserializeVariant(QMetaType::UnknownType, it.value(), object);
262  object->setProperty(qUtf8Printable(it.key()), value);
263  }
264 
265  return object;
266 }

269 {

271  if(!QMetaType::construct(typeId, gadgetPtr, nullptr))
272  throw QJsonDeserializationException(QByteArray("Failed to construct gadget of type ") + QMetaType::typeName(typeId));
273 
274  //now deserialize all json properties
275  for(auto it = jsonObject.constBegin(); it != jsonObject.constEnd(); it++) {
276  auto propIndex = metaObject->indexOfProperty(qUtf8Printable(it.key()));

280  if(property.isEnumType())
281  value = deserializeEnum(property.enumerator(), it.value());

283  value = deserializeVariant(property.userType(), it.value(), nullptr);
