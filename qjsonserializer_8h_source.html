155 template<typename T>
156 bool QJsonSerializer::registerListConverters() {
157  auto ok1 = QMetaType::registerConverter<QList<T>, QVariantList>([](const QList<T> &list) -> QVariantList {
158  QVariantList l;
159  foreach(auto v, list)
160  l.append(QVariant::fromValue(v));
161  return l;
162  });
163 
164  auto ok2 = QMetaType::registerConverter<QVariantList, QList<T>>([](const QVariantList &list) -> QList<T> {
165  QList<T> l;
166  foreach(auto v, list) {
167  auto vt = v.type();
168  if(v.convert(qMetaTypeId<T>()))
169  l.append(v.value<T>());
170  else {
171  qWarning() << QByteArray("Conversion to")
172  << QMetaType::typeName(qMetaTypeId<QList<T>>())
173  << QByteArray("failed, could not convert element of type")
174  << QMetaType::typeName(vt);
175  l.append(T());
176  }
177  }
178  return l;
179  });
180 
181  return ok1 && ok2;
182 }

186 {
187  auto ok1 = QMetaType::registerConverter<QMap<QString, T>, QVariantMap>([](const QMap<QString, T> &map) -> QVariantMap {
188  QVariantMap m;
189  for(auto it = map.constBegin(); it != map.constEnd(); ++it)
190  m.insert(it.key(), QVariant::fromValue(it.value()));
191  return m;
192  });
193 
194  auto ok2 = QMetaType::registerConverter<QVariantMap, QMap<QString, T>>([](const QVariantMap &map) -> QMap<QString, T> {
195  QMap<QString, T> m;
196  for(auto it = map.constBegin(); it != map.constEnd(); ++it) {
197  auto v = it.value();
198  auto vt = v.type();
