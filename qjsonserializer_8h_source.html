128 template<typename T>
129 bool QJsonSerializer::registerListConverters() {
130  auto ok1 = QMetaType::registerConverter<QList<T>, QVariantList>([](const QList<T> &list) -> QVariantList {
131  QVariantList l;
132  foreach(auto v, list)
133  l.append(QVariant::fromValue(v));
134  return l;
135  });
136 
137  auto ok2 = QMetaType::registerConverter<QVariantList, QList<T>>([](const QVariantList &list) -> QList<T> {
138  QList<T> l;
139  foreach(auto v, list) {
140  auto vt = v.type();
141  if(v.convert(qMetaTypeId<T>()))
142  l.append(v.value<T>());
143  else {
144  qWarning() << QByteArray("Conversion to")
145  << QMetaType::typeName(qMetaTypeId<QList<T>>())
146  << QByteArray("failed, could not convert element of type")
147  << QMetaType::typeName(vt);
148  l.append(T());
149  }
150  }
151  return l;
152  });
153 
154  return ok1 && ok2;
155 }
